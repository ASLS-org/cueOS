49 // Creates an empty mailbox.
50 err_t sys_mbox_new(sys_mbox_t *mbox, int size)
51 {
52 #if (osCMSIS < 0x20000U)
53  osMessageQDef(QUEUE, size, void *);
54  *mbox = osMessageCreate(osMessageQ(QUEUE), NULL);
55 #else
56  *mbox = osMessageQueueNew(size, sizeof(void *), NULL);
57 #endif
58 #if SYS_STATS
59  ++lwip_stats.sys.mbox.used;
60  if(lwip_stats.sys.mbox.max < lwip_stats.sys.mbox.used)
61  {
62  lwip_stats.sys.mbox.max = lwip_stats.sys.mbox.used;
63  }
64 #endif /* SYS_STATS */
65  if(*mbox == NULL)
66  return ERR_MEM;
67 
68  return ERR_OK;
69 }

72 /*
73  Deallocates a mailbox. If there are messages still present in the
74  mailbox when the mailbox is deallocated, it is an indication of a
75  programming error in lwIP and the developer should be notified.
76 */
77 void sys_mbox_free(sys_mbox_t *mbox)
78 {
79 #if (osCMSIS < 0x20000U)
80  if(osMessageWaiting(*mbox))
81 #else
82  if(osMessageQueueGetCount(*mbox))
83 #endif
84  {
85  /* Line for breakpoint. Should never break here! */
86  portNOP();
87 #if SYS_STATS
88  lwip_stats.sys.mbox.err++;
89 #endif /* SYS_STATS */
90 
91  }
92 #if (osCMSIS < 0x20000U)
93  osMessageDelete(*mbox);
94 #else
95  osMessageQueueDelete(*mbox);
96 #endif
97 #if SYS_STATS
98  --lwip_stats.sys.mbox.used;
99 #endif /* SYS_STATS */
100 }

140 /*-----------------------------------------------------------------------------------*/
141 /*
142  Blocks the thread until a message arrives in the mailbox, but does
143  not block the thread longer than "timeout" milliseconds (similar to
144  the sys_arch_sem_wait() function). The "msg" argument is a result
145  parameter that is set by the function (i.e., by doing "*msg =
146  ptr"). The "msg" parameter maybe NULL to indicate that the message
147  should be dropped.
148 
149  The return values are the same as for the sys_arch_sem_wait() function:
150  Number of milliseconds spent waiting or SYS_ARCH_TIMEOUT if there was a
151  timeout.
152 
153  Note that a function with a similar name, sys_mbox_fetch(), is
154  implemented by lwIP.
155 */
156 u32_t sys_arch_mbox_fetch(sys_mbox_t *mbox, void **msg, u32_t timeout)
157 {
158 #if (osCMSIS < 0x20000U)
159  osEvent event;
160  uint32_t starttime = osKernelSysTick();
161 #else
162  osStatus_t status;
163  uint32_t starttime = osKernelGetTickCount();
164 #endif
165  if(timeout != 0)
166  {
167 #if (osCMSIS < 0x20000U)
168  event = osMessageGet (*mbox, timeout);
169 
170  if(event.status == osEventMessage)
171  {
172  *msg = (void *)event.value.v;
173  return (osKernelSysTick() - starttime);
174  }
175 #else
176  status = osMessageQueueGet(*mbox, msg, 0, timeout);
177  if (status == osOK)
178  {
179  return (osKernelGetTickCount() - starttime);
180  }
181 #endif
182  else
183  {
184  return SYS_ARCH_TIMEOUT;
185  }
186  }
187  else
188  {
189 #if (osCMSIS < 0x20000U)
190  event = osMessageGet (*mbox, osWaitForever);
191  *msg = (void *)event.value.v;
192  return (osKernelSysTick() - starttime);
193 #else
194  osMessageQueueGet(*mbox, msg, 0, osWaitForever );
195  return (osKernelGetTickCount() - starttime);
196 #endif
197  }
198 }

241 // Creates a new semaphore. The "count" argument specifies
242 // the initial state of the semaphore.
243 err_t sys_sem_new(sys_sem_t *sem, u8_t count)
244 {
245 #if (osCMSIS < 0x20000U)
246  osSemaphoreDef(SEM);
247  *sem = osSemaphoreCreate (osSemaphore(SEM), 1);
248 #else
249  *sem = osSemaphoreNew(UINT16_MAX, count, NULL);
250 #endif
251 
252  if(*sem == NULL)
253  {
254 #if SYS_STATS
255  ++lwip_stats.sys.sem.err;
256 #endif /* SYS_STATS */
257  return ERR_MEM;
258  }
259 
260  if(count == 0) // Means it can't be taken
261  {
262 #if (osCMSIS < 0x20000U)
263  osSemaphoreWait(*sem, 0);
264 #else
265  osSemaphoreAcquire(*sem, 0);
266 #endif
267  }
268 
269 #if SYS_STATS
270  ++lwip_stats.sys.sem.used;
271  if (lwip_stats.sys.sem.max < lwip_stats.sys.sem.used) {
272  lwip_stats.sys.sem.max = lwip_stats.sys.sem.used;
273  }
274 #endif /* SYS_STATS */
275 
276  return ERR_OK;
277 }

279 /*-----------------------------------------------------------------------------------*/
280 /*
281  Blocks the thread while waiting for the semaphore to be
282  signaled. If the "timeout" argument is non-zero, the thread should
283  only be blocked for the specified time (measured in
284  milliseconds).
285 
286  If the timeout argument is non-zero, the return value is the number of
287  milliseconds spent waiting for the semaphore to be signaled. If the
288  semaphore wasn't signaled within the specified time, the return value is
289  SYS_ARCH_TIMEOUT. If the thread didn't have to wait for the semaphore
290  (i.e., it was already signaled), the function may return zero.
291 
292  Notice that lwIP implements a function with a similar name,
293  sys_sem_wait(), that uses the sys_arch_sem_wait() function.
294 */
295 u32_t sys_arch_sem_wait(sys_sem_t *sem, u32_t timeout)
296 {
297 #if (osCMSIS < 0x20000U)
298  uint32_t starttime = osKernelSysTick();
299 #else
300  uint32_t starttime = osKernelGetTickCount();
301 #endif
302  if(timeout != 0)
303  {
304 #if (osCMSIS < 0x20000U)
305  if(osSemaphoreWait (*sem, timeout) == osOK)
306  {
307  return (osKernelSysTick() - starttime);
308 #else
309  if(osSemaphoreAcquire(*sem, timeout) == osOK)
310  {
311  return (osKernelGetTickCount() - starttime);
312 #endif
313  }
314  else
315  {
316  return SYS_ARCH_TIMEOUT;
317  }
318  }
319  else
320  {
321 #if (osCMSIS < 0x20000U)
322  while(osSemaphoreWait (*sem, osWaitForever) != osOK);
323  return (osKernelSysTick() - starttime);
324 #else
325  while(osSemaphoreAcquire(*sem, osWaitForever) != osOK);
326  return (osKernelGetTickCount() - starttime);
327 #endif
328  }
329 }

462 /*
463  This optional function does a "fast" critical region protection and returns
464  the previous protection level. This function is only called during very short
465  critical regions. An embedded system which supports ISR-based drivers might
466  want to implement this function by disabling interrupts. Task-based systems
467  might want to implement this by using a mutex or disabling tasking. This
468  function should support recursive calls from the same task or interrupt. In
469  other words, sys_arch_protect() could be called while already protected. In
470  that case the return value indicates that it is already protected.
471 
472  sys_arch_protect() is only required if your port is supporting an operating
473  system.
474 
475  Note: This function is based on FreeRTOS API, because no equivalent CMSIS-RTOS
476  API is available
477 */
478 sys_prot_t sys_arch_protect(void)
479 {
480 #if (osCMSIS < 0x20000U)
481  osMutexWait(lwip_sys_mutex, osWaitForever);
482 #else
483  osMutexAcquire(lwip_sys_mutex, osWaitForever);
484 #endif
485  return (sys_prot_t)1;
486 }

497 */
